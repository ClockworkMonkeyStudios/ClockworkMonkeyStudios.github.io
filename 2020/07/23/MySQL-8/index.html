MySQL 8.x 安装

MySQL APT Repository

APT 地址：https://dev.mysql.com/downloads/repo/apt/

Ubuntu 内下载命令：wget https://repo.mysql.com//mysql-apt-config_0.8.15-1_all.deb

CREATE DATABASE alpha DEFAULT CHARACTER SET utf8mb4 DEFAULT COLLATE utf8mb4_0900_ai_ci;

CREATE USER ‘alpha’@’%’ IDENTIFIED BY ‘237172.Alpha’;

GRANT ALL PRIVILEGES ON alpha.* TO ‘alpha’@’%’;

FLUSH PRIVILEGES;

exit;
